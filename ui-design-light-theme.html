设计理念

浅色主题保持了深色主题的北欧极简美学，强调色采用较深的青蓝色调，确保在浅色背景上保持良好可读性。背景使用温暖的浅灰而非纯白，提供更舒适的视觉体验。

◈ 实现代码

浏览器自动主题切换的三种实现方式：

方式 1: 使用 prefers-color-scheme (推荐)

@media (prefers-color-scheme: light) {
    :root {
        --color-bg: #f8f8f8;
        --color-fg: #1a1a1a;
        --color-muted: #666666;
        --color-border: #d0d0d0;
        --color-accent: #4a7a8c;
        --color-hover: #e8e8e8;
    }
}

方式 2: 使用 JavaScript 检测系统主题

const mediaQuery = window.matchMedia('(prefers-color-scheme: light)');

function handleThemeChange(e) {
    if (e.matches) {
        document.documentElement.classList.add('light-theme');
    } else {
        document.documentElement.classList.remove('light-theme');
    }
}

mediaQuery.addEventListener('change', handleThemeChange);
handleThemeChange(mediaQuery);

方式 3: 手动切换 + localStorage 持久化

function setTheme(theme) {
    const root = document.documentElement;
    root.classList.remove('light-theme', 'dark-theme');
    
    if (theme === 'light') {
        root.classList.add('light-theme');
    } else if (theme === 'dark') {
        root.classList.add('dark-theme');
    }
    
    localStorage.setItem('theme', theme);
}

function loadTheme() {
    const saved = localStorage.getItem('theme');
    if (saved) {
        setTheme(saved);
    } else {
        // 默认跟随系统
        const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
        setTheme(prefersDark ? 'dark' : 'light');
    }
}

完整实现示例

/* 完整的主题切换 CSS */
:root {
    /* 深色主题 (默认) */
    --color-bg: #0d0d0d;
    --color-fg: #e0e0e0;
    --color-muted: #777777;
    --color-border: #333333;
    --color-accent: #88c0d0;
    --color-hover: #222222;
    
    /* 浅色主题 */
    --light-color-bg: #f8f8f8;
    --light-color-fg: #1a1a1a;
    --light-color-muted: #666666;
    --light-color-border: #d0d0d0;
    --light-color-accent: #4a7a8c;
    --light-color-hover: #e8e8e8;
}

/* 使用 prefers-color-scheme */
@media (prefers-color-scheme: light) {
    :root {
        --color-bg: var(--light-color-bg);
        --color-fg: var(--light-color-fg);
        --color-muted: var(--light-color-muted);
        --color-border: var(--light-color-border);
        --color-accent: var(--light-color-accent);
        --color-hover: var(--light-color-hover);
    }
}

/* 手动切换类 */
:root.light-theme {
    --color-bg: #f8f8f8;
    --color-fg: #1a1a1a;
    --color-muted: #666666;
    --color-border: #d0d0d0;
    --color-accent: #4a7a8c;
    --color-hover: #e8e8e8;
}

/* 过渡效果 */
body {
    transition: background-color 0.3s ease, color 0.3s ease;
}
